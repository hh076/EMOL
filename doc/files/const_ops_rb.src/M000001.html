def query_stepval( step )
    if step == 0 then
        return EMPTY
    elsif step == 1 then
        return UP
    elsif step == 2 then
        return DOWN
    elsif step == 3 then
        return FULL
    else
        $stderr.printf( "Error: query_stepval: 0<= argument < 4 : %s\n", step )
        exit
    end
end
