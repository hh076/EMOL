    def get_total_energy(core_energy, h_mo, eris_mo )
        e = core_energy
        (0..@dens1.size-1).each do | k |
            e += h_mo[ k ] * @dens1[ k ]
#           printf( " %5d   %12.6f  \n", k, @dens1[ k ] )
        end
        (0..@dens2.size-1).each do | k |
            e += eris_mo[ k ] * @dens2[ k ]
        end
        return e
    end
