    def get_hc( c )
        hc = GSL::Matrix.alloc( c.size1, c.size2 )

        @h_diag.each_with_index do | v, i |
            for p in 0...c.size2
               hc[ i, p ] += v * c[ i, p ]
            end
        end

        @index_i.each_with_index do | i, k |
            j = @index_j[ k ] 
            v = @value[ k ]
            for p in 0...c.size2
                hc[ i - 1, p ] += v * c[ j - 1, p ]
                hc[ j - 1, p ] += v * c[ i - 1, p ]
            end
        end
        hc
    end
