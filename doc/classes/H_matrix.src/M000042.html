    def init_vec( _nvec )
        diag_sort = @h_diag.sort
        b = GSL::Matrix.alloc( @n, _nvec )
        for i in 0..._nvec
            index = @h_diag.index( diag_sort[ i ] )
            b[ index, i ] = 1.0
            if index == i 
                b[ mod(i + 1, @n), i ] = 0.00001
            else
                b[ i, i ] = 0.00001
            end
        end
        b
    end
