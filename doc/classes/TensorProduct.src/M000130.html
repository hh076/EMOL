  def hf(iop)
    o = op(iop)
    if ( o == UNITY ) || ( o == CAS ) || ( o == ACS    ) ||
       ( o == CCS   ) || ( o == AAS ) || ( o == CASCAS ) then
      return 1
    elsif ( o == A ) || ( o == C ) || ( o == CCA ) || ( o == CAA ) then
      return 2
    elsif ( o == CAT ) || ( o == ACT ) then
      return 3
    else
      $stderr.printf( "Error: $opstbl::hf: illegal opcode: %s\n", o )
      exit
    end
  end
