    def initialize( _drt )
        @drt = _drt.clone
        csf_list = @drt.mk_csf_list
        @norb = _drt.norb
        @num_coefficient = 0
        expr_all = self.all
        @expr = Expression.new( [@norb, csf_list, expr_all] ) 
        tmp_one = []; one_size = @norb * ( @norb + 1 ) / 2
        expr_all.each do | x |
            if x[0][1] < one_size
                tmp_one.push( x )
            end
        end
        @expr_one = Expression.new( [@norb, csf_list, tmp_one] )
    end
