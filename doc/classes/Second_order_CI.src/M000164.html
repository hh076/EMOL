  def tree_search_csf( inode )
    wk = [ "e", "u", "d", "f" ]
    if inode >= @nodes.size - 1
      @csf_list.push( @csf.clone )
    else
      [0, 1, 2, 3].each do |step|
        next_node = @nodes[ inode ].upper_arc[ step ]
        if next_node != nil
          @walk.push( next_node )
          @csf.push( wk[ step ] )
          tree_search_csf( next_node )
          @walk.pop
          @csf.pop
        end
      end
    end
  end
