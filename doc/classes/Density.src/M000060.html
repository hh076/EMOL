  def initialize( c, nocc, occupation )
    @n = c.size1
    @d = GSL::Matrix.alloc( @n, @n )
    (0..@n-1).each do |p|    
      (0..p).each do |q|
        sum = 0.0
        (0..nocc-1).each do |i|
          sum += c[p,i] * c[q,i]
        end
        @d[p,q] = @d[q,p] = sum * occupation
      end
    end  
  end
