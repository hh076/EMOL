   def initialize( _ncore, _nval, _nexternal )
#           _ncore     : 0-2 電子励起可能な軌道数
#           _nval      : 自由な個数の電子占有可能な軌道数
#           _nexternal : 0-2 電子占有可能な軌道数
      @ncore = _ncore
      @nval  = _nval
      @next  = _nexternal
      @norb  = @ncore + @nval + @next
      @core_range = 1..@ncore
      @val_range = @ncore+1..@ncore+@nval
      @ext_range = @ncore+@nval+1..@ncore+@nval+@next
   end
