    def show
        printf( " CSF List \n" )
        @csf_list.each do | x |
            p x
        end
        printf( " Energy Expression \n" )
        printf( "      IJ       pqrs     Coef \n" )
        ( 0...@ij.size ).each do | i |
            printf( " %7d %10d %14.8f \n", @ij[i], @pqrs[i], @coef[i] )
        end
    end
