    def initialize( obj = nil )
        if obj.nil? then
            @norb = 0
            @csf_list = []
            @ij = []
            @pqrs = []
            @coef = []
        end

        if ( obj.instance_of?( Array ) ) then   #
            @norb = obj[ 0 ]
            @csf_list = obj[ 1 ].clone
            @ij = []
            @pqrs = []
            @coef = []
            obj[2].each do | x |
                @ij.push( x[0][0] + 1 )
                @pqrs.push( x[0][1] )
                @coef.push( x[1].value )
            end
        elsif ( obj.instance_of?( Expression ) ) then
            @norb = obj.norb
            @csf_list = obj.csf_list.clone
            @ij = obj.ij.clone
            @pqrs = obj.pqrs.clone
            @coef = obj.coef.clone 
        else
        end
    end
